def test_toggle_round_trip(tmp_path: Path) -> None:
    """ACT-03: Toggling open_by_default persists through save/load cycle."""
    from joy.store import load_projects, save_projects
    projects_path = tmp_path / "projects.toml"

def action_open_object(self) -> None:
    """Open the highlighted object via operations.open_object (ACT-01, per D-09)."""
    item = self.highlighted_object
    if item is None:
        self.app.notify("No object selected", severity="error", markup=False)
        return
    self._do_open(item)

@work(thread=True, exit_on_error=False)
def _do_open(self, item: ObjectItem) -> None:
    """Run open_object in background thread to avoid blocking TUI."""
    from joy.operations import open_object  # noqa: PLC0415
    try:
        open_object(item=item, config=self.app._config)
        self.app.notify(
            _success_message(item, self.app._config),
            markup=False,
        )
    except Exception:
        from joy.widgets.object_row import _truncate  # noqa: PLC0415
        display = _truncate(item.label if item.label else item.value)
        self.app.notify(f"Failed to open: {display}", severity="error", markup=False)
```

def action_toggle_default(self) -> None:
    """Toggle open_by_default on highlighted object (ACT-03, per D-09, D-12)."""
    item = self.highlighted_object
    if item is None:
        return
    item.open_by_default = not item.open_by_default
    # Update the row's dot indicator in-place
    if 0 <= self._cursor < len(self._rows):
        self._rows[self._cursor].refresh_indicator()
    # Persist in background
    self._save_toggle()

@work(thread=True, exit_on_error=False)
def _save_toggle(self) -> None:
    """Persist toggle change to TOML in background thread (D-12)."""
    from joy.store import save_projects  # noqa: PLC0415
    if hasattr(self.app, "_projects"):
        save_projects(self.app._projects)
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User keystroke -> subprocess | `o` binding dispatches to `operations.open_object()` which calls subprocess |
| Object value -> toast message | User-supplied values from TOML displayed in toast notifications |
| Toggle -> TOML write | `space` modifies in-memory data and writes to disk |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-02-01 | Information Disclosure | action_open_object toast | mitigate | All `app.notify()` calls use `markup=False` to prevent Rich markup injection from user values containing brackets (e.g., branch `[JIRA-123]-fix`). Three explicit `markup=False` usages in this plan. |
| T-03-02-02 | Denial of Service | _do_open worker | mitigate | `@work(thread=True, exit_on_error=False)` prevents subprocess failures from crashing the entire TUI. Exceptions caught and shown as error toasts. |
| T-03-02-03 | Tampering | _save_toggle | accept | Writes to user-owned `~/.joy/projects.toml` via established atomic write pattern. Single-user local tool, no external trust boundary. |
| T-03-02-04 | Elevation of Privilege | open_object subprocess | accept | Already mitigated in Phase 1 (T-1-03-01): AppleScript injection prevented by escaping in `_open_iterm`. Other openers use array-form subprocess.run (no shell=True). |
</threat_model>
