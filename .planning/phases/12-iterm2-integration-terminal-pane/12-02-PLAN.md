```python
class ProjectDetail(Widget, can_focus=True):
    BINDINGS = [
        Binding("escape", "focus_list", "Back"),
        Binding("up", "cursor_up", "Up"),
        Binding("down", "cursor_down", "Down"),
        Binding("k", "cursor_up", "Up"),
        Binding("j", "cursor_down", "Down"),
    ]

    def __init__(self, **kwargs) -> None:
        super().__init__(**kwargs)
        self._cursor: int = -1
        self._rows: list[ObjectRow] = []

    def _update_highlight(self) -> None:
        for row in self._rows:
            row.remove_class("--highlight")
        if 0 <= self._cursor < len(self._rows):
            self._rows[self._cursor].add_class("--highlight")
            self._rows[self._cursor].scroll_visible()

    def action_cursor_up(self) -> None:
        if self._cursor > 0:
            self._cursor -= 1
            self._update_highlight()

    def action_cursor_down(self) -> None:
        if self._cursor < len(self._rows) - 1:
            self._cursor += 1
            self._update_highlight()

    def action_focus_list(self) -> None:
        project_list = self.app.query_one("#project-list")
        listview = project_list.query_one("#project-listview")
        listview.focus()
```

**async set_sessions(sessions: list[TerminalSession] | None) -> None:**
The sole public API for pushing data. Per D-03/D-14/D-17:

1. Save scroll_y from _TerminalScroll
2. `await scroll.remove_children()`
3. If sessions is None: mount `Static("iTerm2 unavailable", classes="empty-state")`, set `_cursor = -1`, `_rows = []`, restore scroll, return
4. If sessions is empty list: mount `Static("No terminal sessions", classes="empty-state")`, set `_cursor = -1`, `_rows = []`, restore scroll, return
5. Split sessions into claude_sessions and other_sessions:
   - Claude: `session.foreground_process == "claude"` (D-08 exact match)
   - Per RESEARCH.md recommendation: ONLY `foreground_process == "claude"` sessions go in Claude group. Sessions where foreground is a shell go in "Other" (no way to detect former-Claude sessions without naming convention).
6. Sort Claude sessions per D-10: busy (foreground=="claude") before waiting (foreground in _SHELL_PROCESSES), then alpha by session_name within each sub-group. Since all Claude sessions have foreground=="claude" (per the filter), they are all "busy" -- sort alphabetically.
7. Sort Other sessions alphabetically by session_name.
8. If claude_sessions: mount `GroupHeader("Claude")`, then SessionRow for each with `is_claude=True, is_busy=True`
9. If other_sessions: mount `GroupHeader("Other")`, then SessionRow for each with `is_claude=False`
10. Build `_rows` list from all mounted SessionRow widgets (in order)
11. Set `_cursor = 0 if _rows else -1`
12. Call `_update_highlight()`
13. Restore scroll_y via `scroll.call_after_refresh(lambda: scroll.scroll_to(y=saved_scroll_y, animate=False))`

```python
def action_focus_session(self) -> None:
    if self._cursor < 0 or self._cursor >= len(self._rows):

    session_id = self._rows[self._cursor].session_id
```
