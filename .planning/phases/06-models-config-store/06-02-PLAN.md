class Repo:
    """A registered git repository."""
    name: str
    local_path: str
    remote_url: str = ""
    forge: str = "unknown"

    def to_dict(self) -> dict:
        return {
            "name": self.name,
            "local_path": self.local_path,
            "remote_url": self.remote_url,
            "forge": self.forge,
        }

def detect_forge(remote_url: str) -> str:
    """Returns 'github', 'gitlab', or 'unknown'."""
    if "github.com" in remote_url:
        return "github"
    if "gitlab.com" in remote_url:
        return "gitlab"
    return "unknown"
```

    def _repos_to_toml(repos: list[Repo]) -> dict:
        """Convert repo list to TOML-serializable dict using keyed schema (D-02)."""
        result: dict = {"repos": {}}
        for repo in repos:
            d = repo.to_dict()
            d.pop("name", None)  # name is the TOML key, not a field inside the table
            result["repos"][repo.name] = d
        return result


    def _toml_to_repos(data: dict) -> list[Repo]:
        """Convert parsed TOML dict to repo list."""
        repos = []
        known_fields = {"local_path", "remote_url", "forge"}
        for name, repo_data in data.get("repos", {}).items():
            for key in repo_data:
                if key not in known_fields:
                    warnings.warn(
                        f"Unknown field {key!r} in repo {name!r} — skipping field",
                        UserWarning,
                        stacklevel=2,
                    )
            repos.append(
                Repo(
                    name=name,
                    local_path=repo_data.get("local_path", ""),
                    remote_url=repo_data.get("remote_url", ""),
                    forge=repo_data.get("forge", "unknown"),
                )
            )
        return repos
    ```

    **Important note on _repos_to_toml:** The `name` field must be REMOVED from the dict before storing because `name` IS the TOML table key (`[repos.joy]`). If "name" is also stored inside the table, it would be redundant. This differs slightly from _projects_to_toml where name IS included inside the table. For repos, follow D-02: name is the key. Pop it from to_dict() output.

    4. Add public functions:

    ```python
    def load_repos(*, path: Path = REPOS_PATH) -> list[Repo]:
        """Load repos from TOML file. Returns empty list if file missing."""
        if not path.exists():
            return []
        with open(path, "rb") as f:
            data = tomllib.load(f)
        return _toml_to_repos(data)


    def save_repos(repos: list[Repo], *, path: Path = REPOS_PATH) -> None:
        """Atomically write repos to TOML file."""
        data = _repos_to_toml(repos)
        content = tomli_w.dumps(data).encode("utf-8")
        _atomic_write(path=path, data=content)


    def get_remote_url(local_path: str) -> str:
        """Get git remote origin URL from a local repo path.

        Per D-07: returns empty string on any error (non-zero exit, timeout, path not found).
        Never raises.
        """
        try:
            result = subprocess.run(
                ["git", "remote", "get-url", "origin"],
                cwd=local_path,
                capture_output=True,
                text=True,
                timeout=5,
            )
            if result.returncode == 0:
                return result.stdout.strip()
            return ""
        except (subprocess.TimeoutExpired, OSError):
            return ""


    def validate_repo_path(local_path: str) -> bool:
        """Check that local_path is an existing directory. Per D-08."""
        return Path(local_path).is_dir()
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| local_path -> subprocess | User-supplied path passed as cwd to git subprocess |
| TOML file -> application | repos.toml content parsed into Repo objects |
| local_path -> filesystem | Path string checked via Path.is_dir() |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-03 | Tampering | get_remote_url() subprocess | mitigate | Use list form `["git", "remote", "get-url", "origin"]` — never shell=True. cwd is the only user-controlled parameter and it goes to the OS, not a shell. timeout=5 prevents hangs. |
| T-06-04 | Information Disclosure | repos.toml file permissions | accept | File created with user's default umask under ~/.joy/. Same security model as projects.toml and config.toml already in production. No secrets stored (paths and URLs only). |
| T-06-05 | Denial of Service | get_remote_url() timeout | mitigate | timeout=5 on subprocess.run prevents hanging on unresponsive git operations. OSError caught for invalid paths. |
| T-06-06 | Tampering | _toml_to_repos() unknown fields | mitigate | Unknown fields emit warnings.warn and are skipped — no arbitrary data injected into Repo objects. Only known_fields {"local_path", "remote_url", "forge"} are read. |
| T-06-07 | Spoofing | validate_repo_path() | accept | Path.is_dir() checks filesystem state. Symlink following is acceptable — if user symlinks a repo path, that's intentional. No TOCTOU concern because validation is advisory (path could be removed between check and use). |
</threat_model>
