   ```python
   def action_new_project(self) -> None:
       """Start project creation flow: name modal then add-object loop (D-01, D-02)."""
       def on_name(name: str | None) -> None:
           if name is None:
               return
           # D-04: Check duplicate name
           if any(p.name == name for p in self._projects):
               self.notify(f"Project '{name}' already exists", severity="error", markup=False)
               return
           # Create project, add to list, persist, refresh
           project = Project(name=name)
           self._projects.append(project)
           self._save_projects_bg()
           project_list = self.query_one(ProjectList)
           project_list.set_projects(self._projects)
           # Select the new project (last in list)
           listview = project_list.query_one("#project-listview", JoyListView)
           listview.index = len(self._projects) - 1
           self.query_one(ProjectDetail).set_project(project)
           self.notify(f"Created project: '{name}'", markup=False)
           # D-02, D-03: Start add-object loop
           self._start_add_object_loop(project)
       self.push_screen(NameInputModal(), on_name)
   ```

4. Add `_start_add_object_loop()` method to JoyApp:
   ```python
   def _start_add_object_loop(self, project: Project) -> None:
       """Loop: preset picker -> value input -> repeat until Escape (D-03)."""
       def on_preset(preset: PresetKind | None) -> None:
           if preset is None:
               return  # Escape exits loop
           def on_value(value: str | None) -> None:
               if value is not None:
                   obj = ObjectItem(kind=preset, value=value)
                   project.objects.append(obj)
                   self._save_projects_bg()
                   self.query_one(ProjectDetail).set_project(project)
                   self.notify(f"Added: {preset.value} '{_truncate(value)}'", markup=False)
               # Loop: push preset picker again regardless of value result (D-03)
               self._start_add_object_loop(project)
           self.push_screen(ValueInputModal(preset), on_value)
       self.push_screen(PresetPickerModal(), on_preset)
   ```

@pytest.mark.asyncio
async def test_n_creates_project(mock_store, mock_save):
    """PROJ-04: Pressing n, typing name, Enter creates a new project."""
    app = JoyApp()
    async with app.run_test() as pilot:
        await pilot.pause(0.2)
        await app.workers.wait_for_complete()
        initial_count = len(app._projects)

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User Input -> Project list | User-typed project name becomes a key in projects.toml and appears in ListView |
| User Input -> Object value | User-typed value stored as ObjectItem.value in projects.toml |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-4-06 | Tampering | action_new_project | mitigate | Duplicate project name check (`any(p.name == name for p in self._projects)`) prevents silent data overwrite. tomli_w escapes special TOML chars in project name. |
| T-4-07 | Denial of Service | _start_add_object_loop | accept | Recursive push_screen loop could theoretically stack frames, but user must actively press Enter each time. No unbounded recursion risk. |
| T-4-08 | Injection | notify() | mitigate | All notify() calls use markup=False per project convention. User-supplied project names and values never interpreted as Rich markup. |
| T-4-09 | Tampering | _save_projects_bg | accept | Concurrent saves are benign due to atomic write (os.replace). Last write wins, single user, fast saves. |
</threat_model>
