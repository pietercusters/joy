def close_session(session_id: str, force: bool = False) -> bool:
    import iterm2
    from iterm2.connection import Connection
    success = False
    async def _close(connection):
        nonlocal success
        app = await iterm2.async_get_app(connection)
        session = app.get_session_by_id(session_id)
        if session is None:
            success = True
            return
        await session.async_close(force=force)
        success = True
    try:
        Connection().run_until_complete(_close, retry=False)
    except Exception:
        pass
    return success
```

def close_tab(tab_id: str, force: bool = False) -> bool:
    """Close an entire iTerm2 tab by tab_id. Returns True on success, False on failure.

    force=False for graceful close, force=True to skip iTerm2's confirmation.
    If tab is already gone (not found in any window), returns True.
    All iterm2 imports are lazy to avoid startup overhead.
    """
    import iterm2
    from iterm2.connection import Connection

    success = False

    async def _close(connection):
        nonlocal success
        app = await iterm2.async_get_app(connection)
        for window in app.terminal_windows:
            for tab in window.tabs:
                if tab.tab_id == tab_id:
                    await tab.async_close(force=force)
                    success = True
                    return
        # Tab not found in any window — already gone
        success = True

    try:
        Connection().run_until_complete(_close, retry=False)
    except Exception:
        pass
    return success
```

Key implementation details:
- Lazy `import iterm2` and `from iterm2.connection import Connection` inside the function body (not at module level) per established pattern
- Iterates `app.terminal_windows` -> `window.tabs` to find the tab by `tab.tab_id` (same loop structure as `fetch_sessions` lines 80-82)
- Calls `tab.async_close(force=force)` (iTerm2 Tab object has this method per D-07 specifics in CONTEXT.md)
- If no matching tab found: sets `success = True` (already gone -- mirrors the `session is None` guard in close_session)
- try/except around `Connection().run_until_complete` catches any iTerm2 connection failure -- returns False (silent fail per established contract)
