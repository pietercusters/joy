def _archived_to_toml(archived_projects: list[ArchivedProject]) -> dict:
    """Convert archived project list to TOML-serializable dict (keyed schema).

    Top-level key is 'archive' to distinguish from projects.toml.
    archived_at stored as offset-aware datetime — tomli_w serializes natively.
    """
    result: dict = {"archive": {}}
    for ap in archived_projects:
        d = ap.project.to_dict()
        d["archived_at"] = ap.archived_at  # datetime with tzinfo — tomli_w handles it
        result["archive"][ap.project.name] = d
    return result


def _toml_to_archived(data: dict) -> list[ArchivedProject]:
    """Convert parsed archive.toml dict to ArchivedProject list."""
    archived = []
    for name, entry in data.get("archive", {}).items():
        # Parse the inner project (reuse _toml_to_projects logic inline)
        objects = []
        for obj in entry.get("objects", []):
            raw_kind = obj.get("kind", "")
            try:
                kind = PresetKind(raw_kind)
            except ValueError:
                warnings.warn(
                    f"Unknown object kind {raw_kind!r} in archived project {name!r} — skipping",
                    UserWarning,
                    stacklevel=2,
                )
                continue
            value = obj.get("value")
            if value is None:
                warnings.warn(
                    f"Object in archived project {name!r} missing 'value' — skipping",
                    UserWarning,
                    stacklevel=2,
                )
                continue
            objects.append(
                ObjectItem(
                    kind=kind,
                    value=value,
                    label=obj.get("label", ""),
                    open_by_default=obj.get("open_by_default", False),
                )
            )
        created_raw = entry.get("created")
        if isinstance(created_raw, date):
            created = created_raw
        elif isinstance(created_raw, str):
            try:
                created = date.fromisoformat(created_raw)
            except ValueError:
                created = date.today()
        else:
            created = date.today()

        archived_at_raw = entry.get("archived_at")
        if isinstance(archived_at_raw, datetime):
            archived_at = archived_at_raw
            # Ensure timezone-aware (tomllib returns offset-aware for TOML offset datetimes)
            if archived_at.tzinfo is None:
                archived_at = archived_at.replace(tzinfo=timezone.utc)
        else:
            archived_at = datetime.now(timezone.utc)

        project = Project(
            name=name,
            objects=objects,
            created=created,
            repo=entry.get("repo"),
        )
        archived.append(ArchivedProject(project=project, archived_at=archived_at))
    return archived


def load_archived_projects(*, path: Path = ARCHIVE_PATH) -> list[ArchivedProject]:
    """Load archived projects from archive.toml. Returns [] if file missing."""
    if not path.exists():
        return []
    with open(path, "rb") as f:
        data = tomllib.load(f)
    return _toml_to_archived(data)


def save_archived_projects(
    archived_projects: list[ArchivedProject], *, path: Path = ARCHIVE_PATH
) -> None:
    """Atomically write archived projects to archive.toml."""
    data = _archived_to_toml(archived_projects)
    content = tomli_w.dumps(data).encode("utf-8")
    _atomic_write(path=path, data=content)
```

"""ArchiveModal: three-option confirmation dialog for archiving a project."""
from __future__ import annotations

from enum import Enum

from textual.app import ComposeResult
from textual.containers import Vertical
from textual.screen import ModalScreen
from textual.widgets import Static

from joy.models import Project


class ArchiveChoice(str, Enum):
    """Result of the ArchiveModal — what the user chose to do."""

    ARCHIVE_WITH_CLOSE = "archive_with_close"   # Enter: archive + close iTerm2 sessions
    ARCHIVE_ONLY = "archive_only"               # a: archive, keep sessions running
    CANCEL = "cancel"                           # Esc: do nothing


class ArchiveModal(ModalScreen["ArchiveChoice"]):
    """Modal to confirm archiving a project.

    Three choices:
      Enter → ARCHIVE_WITH_CLOSE (archive and close iTerm2 sessions)
      a     → ARCHIVE_ONLY       (archive, leave sessions running)
      Esc   → CANCEL
    """

    BINDINGS = [
        ("escape", "cancel", "Cancel"),
        ("enter", "confirm_with_close", "Archive + Close sessions"),
        ("a", "confirm_only", "Archive only"),
    ]

    DEFAULT_CSS = """
    ArchiveModal {
        align: center middle;
    }
    ArchiveModal > Vertical {
        width: 64;
        height: auto;
        background: $surface;
        border: thick $warning;
        padding: 1 2;
    }
    ArchiveModal .modal-title {
        text-style: bold;
    }
    ArchiveModal .modal-hint {
        color: $text-muted;
    }
    """

    def __init__(self, project: Project) -> None:
        super().__init__()
        self._project = project

    def compose(self) -> ComposeResult:
        with Vertical():
            yield Static(f"Archive: {self._project.name}", classes="modal-title")
            yield Static(
                "Worktree and terminal objects will be stripped. All other objects preserved."
            )
            yield Static(
                "Enter: archive + close sessions  •  a: archive only  •  Esc: cancel",
                classes="modal-hint",
            )

    def on_mount(self) -> None:
        self.focus()

    def action_confirm_with_close(self) -> None:
        self.dismiss(ArchiveChoice.ARCHIVE_WITH_CLOSE)

    def action_confirm_only(self) -> None:
        self.dismiss(ArchiveChoice.ARCHIVE_ONLY)

    def action_cancel(self) -> None:
        self.dismiss(ArchiveChoice.CANCEL)

        width: 70;
        max-height: 80%;
        background: $surface;
        border: solid $accent;
        padding: 1 2;
    }
    ArchiveBrowserModal .modal-title {
        text-style: bold;
    }
    ArchiveBrowserModal .modal-hint {
        color: $text-muted;
        margin-top: 1;
    }
    """

    def __init__(self, archived: list[ArchivedProject], active_branches: set[str]) -> None:
        super().__init__()
        self._archived = archived
        self._active_branches = active_branches
        self._rows: list[_ArchiveRow] = []
        self._cursor: int = -1

    def compose(self) -> ComposeResult:
        with Vertical():
            yield Static("Archived Projects", classes="modal-title")
            yield VerticalScroll(id="archive-scroll")
            yield Static(
                "up/down: navigate  •  u: unarchive  •  Esc: close",
                classes="modal-hint",
            )

    def on_mount(self) -> None:
        self._rebuild()
        self.focus()

    def _partition_by_branch(
        self,
    ) -> tuple[list[ArchivedProject], list[ArchivedProject]]:
        """Split archived projects into branch-matched and rest, both descending."""
        matched: list[ArchivedProject] = []
        rest: list[ArchivedProject] = []
        for ap in self._archived:
            branch_values = {
                obj.value
                for obj in ap.project.objects
                if obj.kind == PresetKind.BRANCH
            }
            if branch_values & self._active_branches:
                matched.append(ap)
            else:
                rest.append(ap)
        matched.sort(key=lambda ap: ap.archived_at, reverse=True)
        rest.sort(key=lambda ap: ap.archived_at, reverse=True)
        return matched, rest

    def _rebuild(self) -> None:
        """Mount section headers and rows into the scroll container."""
        scroll = self.query_one("#archive-scroll", VerticalScroll)
        scroll.remove_children()
        self._rows = []

        matched, rest = self._partition_by_branch()

        if matched:
            scroll.mount(_SectionHeader("Active Branch"))
            for ap in matched:
                row = _ArchiveRow(ap)
                scroll.mount(row)
                self._rows.append(row)

        if rest:
            scroll.mount(_SectionHeader("Other"))
            for ap in rest:
                row = _ArchiveRow(ap)
                scroll.mount(row)
                self._rows.append(row)

        self._cursor = 0 if self._rows else -1
        self._update_highlight()

    def _update_highlight(self) -> None:
        for row in self._rows:
            row.remove_class("--highlight")
        if 0 <= self._cursor < len(self._rows):
            self._rows[self._cursor].add_class("--highlight")
            self._rows[self._cursor].scroll_visible()

    def action_cursor_up(self) -> None:
        if self._cursor > 0:
            self._cursor -= 1
            self._update_highlight()

    def action_cursor_down(self) -> None:
        if self._cursor < len(self._rows) - 1:
            self._cursor += 1
            self._update_highlight()

    def action_unarchive(self) -> None:
        if 0 <= self._cursor < len(self._rows):
            self.dismiss(self._rows[self._cursor].archived)
        else:
            self.dismiss(None)

    def action_cancel(self) -> None:
        self.dismiss(None)
```

def action_archive_project(self) -> None:
    """Archive the highlighted project to archive.toml (removes worktree/terminal objects)."""
    from joy.screens.archive_modal import ArchiveChoice, ArchiveModal  # noqa: PLC0415

    if self._cursor < 0 or self._cursor >= len(self._rows):
        return
    project = self._rows[self._cursor].project
    cursor_at = self._cursor

    def on_archive(choice: ArchiveChoice) -> None:
        if choice is ArchiveChoice.CANCEL:
            return

        close_terminals = choice is ArchiveChoice.ARCHIVE_WITH_CLOSE

        # Strip WORKTREE + TERMINALS; preserve all other objects
        stripped_objects = [
            obj for obj in project.objects
            if obj.kind not in (PresetKind.WORKTREE, PresetKind.TERMINALS)
        ]

        from joy.models import Project as _Project  # noqa: PLC0415
        archived_project_data = _Project(
            name=project.name,
            objects=stripped_objects,
            created=project.created,
            repo=project.repo,
        )

        # Optionally close terminal sessions in background
        if close_terminals and self.app._rel_index is not None:
            sessions = self.app._rel_index.terminals_for(project)
            if sessions:
                self.app._close_sessions_bg(sessions)

        # Remove from active projects and persist
        projects = self.app._projects
        try:
            projects.remove(project)
        except ValueError:
            return
        self.app._save_projects_bg()
        self.set_projects(projects, self._repos)

        # Add to archive in background
        from datetime import datetime, timezone  # noqa: PLC0415
        from joy.models import ArchivedProject  # noqa: PLC0415
        archived = ArchivedProject(
            project=archived_project_data,
            archived_at=datetime.now(timezone.utc),
        )
        self.app._append_to_archive_bg(archived)

        # Restore cursor
        if projects:
            new_index = min(cursor_at, len(projects) - 1)

            def _restore() -> None:
                self.focus()
                self.select_index(new_index)

            self.call_after_refresh(_restore)

        self.app.notify(f"Archived: '{project.name}'", markup=False)

    self.app.push_screen(ArchiveModal(project=project), on_archive)

def action_open_archive_browser(self) -> None:
    """Open the archive browser popup (A key) to browse and unarchive projects."""
    from joy.screens.archive_browser import ArchiveBrowserModal  # noqa: PLC0415
    from joy.store import load_archived_projects  # noqa: PLC0415

    archived = load_archived_projects()
    if not archived:
        self.app.notify("No archived projects.", markup=False)
        return

    active_branches: set[str] = {
        wt.branch
        for wt in self.app._current_worktrees
        if wt.branch and wt.branch != "HEAD"
    }

    def on_unarchive(result: object) -> None:
        from joy.models import ArchivedProject as _AP  # noqa: PLC0415
        if result is None or not isinstance(result, _AP):
            return
        # Restore project (already stripped of WORKTREE/TERMINALS at archive time)
        self.app._projects.append(result.project)
        self.app._save_projects_bg()
        self.app._remove_from_archive_bg(result)
        self.set_projects(list(self.app._projects), self._repos)
        # Move cursor to the restored project
        restored_name = result.project.name

        def _find_restored() -> None:
            for i, row in enumerate(self._rows):
                if row.project.name == restored_name:
                    self.focus()
                    self.select_index(i)
                    break

        self.call_after_refresh(_find_restored)
        self.app.notify(f"Unarchived: '{result.project.name}'", markup=False)

    self.app.push_screen(
        ArchiveBrowserModal(archived=archived, active_branches=active_branches),
        on_unarchive,
    )

@work(thread=True, exit_on_error=False)
def _close_sessions_bg(self, sessions: list) -> None:
    """Close iTerm2 sessions in background thread.

    Accepts list[TerminalSession]. Typed as list to avoid circular import at
    module level; runtime objects always have .session_id: str.
    """
    from joy.terminal_sessions import close_session  # noqa: PLC0415
    for session in sessions:
        close_session(session.session_id, force=False)

@work(thread=True, exit_on_error=False)
def _append_to_archive_bg(self, archived: object) -> None:
    """Load archive.toml, append new entry, save atomically (background thread).

    Note: not safe for concurrent calls — single-user TUI only.
    """
    from joy.store import load_archived_projects, save_archived_projects  # noqa: PLC0415
    from joy.models import ArchivedProject  # noqa: PLC0415
    assert isinstance(archived, ArchivedProject)
    existing = load_archived_projects()
    existing.append(archived)
    save_archived_projects(existing)

@work(thread=True, exit_on_error=False)
def _remove_from_archive_bg(self, archived: object) -> None:
    """Load archive.toml, remove the given entry, save atomically (background thread).

    Matches by project name. Note: not safe for concurrent calls.
    """
    from joy.store import load_archived_projects, save_archived_projects  # noqa: PLC0415
    from joy.models import ArchivedProject  # noqa: PLC0415
    assert isinstance(archived, ArchivedProject)
    existing = load_archived_projects()
    updated = [ap for ap in existing if ap.project.name != archived.project.name]
    save_archived_projects(updated)
